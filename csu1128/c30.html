This program prints the triangle of numbers of the range entered by the user. Example 3 = 1 2 3 2 1 

                    
#include <stdio.h>
int main()
{
    int d, m, j, o, n;
    printf("\n\n This program prints the triangle of numbers of the range entered by the user. Example 3 = 1 2 3 2 1 \n\n");

    printf("Enter a number for which you want to print its triangle and its reverse (Example: 4 3 2 1 2 3 4) - and press enter: ");
    scanf("%d", &n);
    printf("\n Triangle for the entered range is %d is: \n", n);
    for (d = 1; d <= n; d++)
    {
        for (m = 1; m <= n - d; m++)
        {
            printf("  ");
        }
        for (j = 1; j <= d; j++)
        {
            printf("%d ", j);
        }
        for (o = d - 1; o >= 1; o--)
        {
            printf("%d ", o);
        }
        printf("\n");
    }
    return 0;
}

Output:

1
1 2 1
1 2 3 2 1
1 2 3 4 3 2 1
