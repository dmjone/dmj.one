This program prints the triangle of numbers of the range entered by the user. Example 3 = 1 2 3 2 1 

                    
#include <stdio.h>
int main()
{
    // declare the local variables
    int i, j, k=0, p;
    printf("Enter a number: ");
    scanf("%d", &p);
    int q = p+1;
        for (i = 0; i <= p; i++)
        {
            for (j = 1; j <= p-i; j++)
            {
                printf("  ");
            }
            for (k = 1; k <=i; k++)
            {
                printf(" %d", q-k);
            }
            for(int l=p-i;l<=p;l++)
            {
                printf(" %d",l);

            }
            printf ("\n");
        }
    return 0;
}
